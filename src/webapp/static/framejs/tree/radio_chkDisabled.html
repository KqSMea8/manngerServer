	{ id:111, pId:11, name:"disabled 1-1-1", chkDisabled:true, checked:true},

function disabledNode(e) {
	disabled = e.data.disabled,
	nodes = treeDemo.getSelectedNodes();
	if (nodes.length == 0) {
		alert("请先选择一个节点");
	}

	for (var i=0, l=nodes.length; i<l; i++) {
		treeDemo.setChkDisabled(nodes[i], disabled);
	}
}
var setting__radio_chkDisabled = {
    id:"#treeDemo_radio_chkDisabled",
    nodes:zNodes,  //数据节点指定
	check: {
		enable: true,
		chkStyle: "radio",
		radioType: "level"
	},
	data: {
		simpleData: {
			enable: true
		}
	}
};
var treeDemo;
$(document).ready(function(){
	treeDemo =  new biz.tree(setting__radio_chkDisabled);//创建树
	$("#disabledTrue").bind("click", {disabled: true}, disabledNode);
	$("#disabledFalse").bind("click", {disabled: false}, disabledNode);

});
