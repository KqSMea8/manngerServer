function setEdit() {
	var remove = $("#remove").attr("checked");
	var rename = $("#rename").attr("checked");
	var removeTitle = $.trim($("#removeTitle").get(0).value);
	var renameTitle = $.trim($("#renameTitle").get(0).value);
	treeDemo.setting.edit.showRemoveBtn = remove;
	treeDemo.setting.edit.showRenameBtn = rename;
	treeDemo.setting.edit.removeTitle = removeTitle;
	treeDemo.setting.edit.renameTitle = renameTitle;
	showCode(['setting.edit.showRemoveBtn = ' + remove, 'setting.edit.showRenameBtn = ' + rename,
		'setting.edit.removeTitle = "' + removeTitle +'"', 'setting.edit.renameTitle = "' + renameTitle + '"']);
}
function showCode(str) {
	var code_edit = $("#code_edit");
	code_edit.empty();
	for (var i=0, l=str.length; i<l; i++) {
		code_edit.append("<li>"+str[i]+"</li>");
	}
}
var setting_edit = {
    id:"#treeDemo_edit",
    nodes:zNodes,  //数据节点指定
	edit: {
		enable: true
	},
	data: {
		simpleData: {
			enable: true
		}
	},
	callback: {
		beforeDrag: beforeDrag
	}
};
var treeDemo;
$(document).ready(function(){
	treeDemo =  new biz.tree(setting_edit);//创建树
	setEdit();
	$("#remove").bind("change", setEdit);
	$("#rename").bind("change", setEdit);
	$("#removeTitle").bind("propertychange", setEdit)
	.bind("input", setEdit);
	$("#renameTitle").bind("propertychange", setEdit)
	.bind("input", setEdit);
});
