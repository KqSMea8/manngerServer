示例

    • 1、setting 配置信息说明

      • 使用 checkbox，必须设置 setting.check 中的各个属性

      • 父子关联关系：
        被勾选时：关联父 关联子
        取消勾选时：关联父 关联子

      • 2、treeNode 节点数据说明

        • 1)、如果需要初始化默认节点被勾选，请设置 treeNode.checked 属性
        • 2)、如果某节点禁用 checkbox，请设置 treeNode.chkDisabled 属性，详细请参见'chkDisabled 演示'
        • 3)、如果某节点不显示 checkbox，请设置 treeNode.nocheck 属性，详细请参见'nocheck 演示'

var code;

function setCheck() {
	var py = $("#py").attr("checked")? "p":"";

	var sy = $("#sy").attr("checked")? "s":"";
	var pn = $("#pn").attr("checked")? "p":"";
	var sn = $("#sn").attr("checked")? "s":"";
	var type = { "Y":py + sy, "N":pn + sn};
	treeDemo.setting.check.chkboxType = type;
	showCode('setting.check.chkboxType = { "Y" : "' + type.Y + '", "N" : "' + type.N + '" };');
}
function showCode(str) {
	if (!code) code = $("#code");
	code.empty();
	code.append("<li>"+str+"</li>");
}
var setting_checkbox = {
    id:"#treeDemo_checkbox",
    nodes:zNodes,  //数据节点指定
	check: {
		enable: true
	},
	data: {
		simpleData: {
			enable: true
		}
	}
};
var treeDemo;
$(document).ready(function(){
	treeDemo =  new biz.tree(setting_checkbox);//创建树
	setCheck();
	$("#py").bind("change", setCheck);
	$("#sy").bind("change", setCheck);
	$("#pn").bind("change", setCheck);
	$("#sn").bind("change", setCheck);
});
