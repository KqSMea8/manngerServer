new biz.button({id:"#alert_button",label: "信息提示" }).click(function() {
   new biz.alert({type:"alert",message:"这是一个信息提醒对话框！",title:"注 意"}) ;
   //调用alert组件，type：类型，分为alert、confirm、prompt
   //message:提示信息
   //titile：提示框标题
});

 
js部分:
new biz.button({id:"#hidenauto_button",label: "信息提示" }).click(function() {
	new biz.alert({type:"alert",message:"我会在2秒后自动消失！",title:"注 意"}) ;
	new biz.alert({type:"hide",times:2000}) ;//调用hide命令，times：持久时间
});
 
js部分:
new biz.button({id:"#confirm_button",label: "信息确认" }).click(function() {
	   var msg = "确定执行操作?",tit="确认信息提示",
	   fuc =  function(r) {
		  new biz.alert({type:"alert",message:'确定信息: ' + r,title:"提示"}) ;
	   }
	   new biz.alert({type:"confirm",message:msg,title:tit,callback:fuc}) ;
	   //callback：回调函数，参数r，true（点击了“确定”）或false（点击了“取消”）
});
 
js部分:
new biz.button({id:"#prompt_button",label: "输入提示" }).click(function() {
        new biz.alert({type:"prompt",message:"输入信息:",value:'请在这输入..',title:"信息输入提示",
             callback:function(r) {if( r ) alert('您输入了： ' + r);}
         }) ;
		//value:输入框默认显示内容
		//callback：回调函数。参数r：输入的内容
});
